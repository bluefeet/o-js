o.js

Javascript utility belt for rich objects and prototypes. Ooooh!

Why o.js?

JavaScript already has built-in support for objects and inheritance, but it is very lean and only provides the most basic functionality. This library aims to fill in missing the pieces.

Flexible

There are several libraries already that provide rich OO frameworks for JavaScript. This library does not provide a framework, and does not impose standard OO. Use it as you see fit!

Simple

One of the tenants of great software design is "do one thing, and do it well". We follow this phylosophy by providing well thought-out bite-sized features which work independently and compliment each other when used together.

Constructor

The classic point example, o.js style, using o.construct and o.accessor.

var Point = o.construct(
    function (x, y) {
        this.x( x );
        this.y( y );
    },
    {
        x: o.accessor('_x', { isa:'number' }),
        y: o.accessor('_x', { isa:'number' }),
        distanceTo: function (other) {
            xd = this.x() - other.x();
            yd = this.y() - other.y();
            return Math.sqrt(xd*xd + yd*yd)
        }
    }
);

var a = new Point( 10, 32 );
var b = new Point( 4, 5 );

console.log( o.distanceTo(b) ); // 27.658...

Inheritance

This literally extends the point example by adding a third dimension, using o.extend.

var Point3D = o.extend(
    Point,
    function (parent, x, y, z) {
        parent( x, y );
        this.z( z );
    },
    {
        z: o.accessor('_z', { isa:'number' }),
        distanceTo = function(other) {
            xd = this.x() - other.x();
            yd = this.y() - other.y();
            zd = this.z() - other.z();
            return Math.sqrt(xd*xd + yd*yd + zd*zd)
        }
    }
);

var a = new Point3D( 10, 32, 5 );
var b = new Point3D( 4, 5, 15 );

console.log( o.distanceTo(b) ); // 29.410...
